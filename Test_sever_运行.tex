%Introduction
\textrm{Test\_sever.py~}是本人根据\textrm{Pymatgen~}编写的针对清华大学物理系王先生的作业提交脚本。

运行\textrm{Test\_sever.py}:
\begin{enumerate}
	\item 配置运行\textrm{Python~}环境\\
		编辑\textrm{Test\_sever.py~}文本的第一行，指定\textrm{Python~}命令位置:\\
		\textcolor{green}{\#!\underline{\$Python\_Installed}}\\例如当前无锡超算\textrm{duzhh}账号下的设置为\\
		\textrm{\#!/home/export/online1/tsinghua/duzhh/Softs/Python-2.7.11/bin/python}
	\item 查看\textrm{Test\_sever.py~}帮助命令\\
		 \textcolor{red}{\textrm{Test\_sever.py\;\;-h}}
\begin{figure}[h!]
\centering
\includegraphics{Test_sever-h.png}
%\caption{\small Band Structure of PbTe (a) and EuTe (b).}%(与文献\cite{EPJB33-47_2003}图1对比)
\label{Test_sever.py-help}
\end{figure}
	 \item 当前\textrm{Test\_sever.py~}运行需要的文件\\
		 \textcolor{violet}{POSCAR}$\leftarrow$原始\textrm{Ni-}基单晶高温合金$\textrm{Ni}_3\textrm{Al}$未掺杂(母体)模型\\
		 \textcolor{violet}{Replace\_Elements}$\leftarrow$合金掺杂元素(当前算例中包含指定的19种元素)
\begin{figure}[h!]
\centering
\includegraphics{Test_sever-files.png}
%\caption{\small Band Structure of PbTe (a) and EuTe (b).}%(与文献\cite{EPJB33-47_2003}图1对比)
\label{Test_sever.py-files}
\end{figure}
	 \item 当前\textrm{Test\_sever.py~}的运行:\\
		 \textcolor{red}{./Test\_sever.py\;\;-N}\; \textcolor{blue}{Num}\\
		 \textcolor{blue}{Num}指定掺杂合金模型类型数(当前算例参见王先生模型) \\
		 \textcolor{magenta}{王先生共设计了若干组模型，这里给出其中两组，每组都是六个模型}\\
\begin{figure}[h!]
\centering
\includegraphics[height=3.35in,width=5.35in,viewport=0 0 1250 800,clip]{03_NiAl-Ni_single.png}
\caption{\small 第一组6个模型，每个模型分别取代一个原子.}%(与文献\cite{EPJB33-47_2003}图1对比)
\label{Wang_Model_1}
\end{figure}

\begin{figure}[h!]
\centering
\includegraphics[height=3.35in,width=5.35in,viewport=0 0 1250 800,clip]{04_NiAl-Ni_N-N.png}
\caption{\small 第二组6个模型，每个模型分别取代两个原子.}%(与文献\cite{EPJB33-47_2003}图1对比)
\label{Wang_Model_2}
\end{figure}
	 	 执行\textrm{Test\_sever.py}的提示与输入:\\
		 \textcolor{violet}{Give The Num. of Added\_Element(s) AND Relapce\_Num :\\1. Atom\_Num1,Atom\_Num2,Atom\_Num3 \dots}\\
		 \textcolor{blue}{输入:~取代原子的数目(依次按原子顺序)}\\
		 \textcolor{violet}{2. The Index of the Replace-Position\\ Index=}\\
		 \textcolor{blue}{输入:~取代原子的位置}\\
		 \textcolor{violet}{Alloy\_\$\textbf{Metal}\\
		  The XC of POTCAR generate by: \\
	   The model of POTCAR for this ELEMENT (sv/pv/h\dots) :}\\
	   \textcolor{blue}{指定确定元素的势函数类型}$\leftarrow$默认为\textrm{PBE~}型赝势\\
	   可根据变量\textcolor{orange}{\$PMG\_VASP\_PSP\_DIR~确定的目录下的赝势名称确定}\\
	   \textcolor{red}{本算例中，只有元素\textrm{Nb}需要给定参数\textbf{pv}，其余直接输入“enter”即可}
\begin{figure}[h!]
\centering
\includegraphics{Test_sever-Input.png}
\caption{\small 第二组6个模型中第一个模型的输入.}%(与文献\cite{EPJB33-47_2003}图1对比)
\label{Test_sever.py-Input}
\end{figure}
\end{enumerate}

\textbf{\textcolor{red}{{注意:~}为了正确运行脚本，请在脚本中指定正确的执行命令}}
